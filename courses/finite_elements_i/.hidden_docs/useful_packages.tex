    discard if larger/.style 2 args={
        x filter/.code={
          \edef\tempa{\thisrow{#1}}
          \edef\tempb{#2}
					\ifnum\tempa>\tempb
					\def\pgfmathresult{inf}
					\else\fi}
			}
		}

    discard if not/.style 2 args={
        x filter/.code={
            \edef\tempa{\thisrow{#1}}

            \else
                \def\pgfmathresult{inf}
            \fi
        }
    }
}
